//@version=6
// =================================================================
// BAMBAM 24-Feature Delta Export v2.2 — TRADINGVIEW PINE SCRIPT
// =================================================================
//
// PURPOSE:
//   Exports every raw BAMBAM signal as a Strategy Tester trade entry.
//   Each entry embeds 24 computed features as a pipe-delimited string
//   in the comment field of strategy.entry(). When you Export CSV from
//   the Strategy Tester, the Signal column contains:
//
//     BEAR|f0|f1|...|f23
//     BULL|f0|f1|...|f23
//
//   The downstream Python parser (enrich_signals.py) splits on '|' and
//   expects exactly 25 pipe-delimited parts per row (1 direction + 24
//   floats). Number of parts != 25 → row is skipped.
//
// BUG HISTORY (read before auditing):
//
//   v1 (original, May 22 2026):
//     Working version used bare str.tostring(val) with plain pipe-
//     delimited floats. But volRatio_delta (feature 19) used a
//     shifted averaging window, comparing mismatched denominators.
//     Created systematic bias: v1 volRatio_delta mean=1.61, 70%
//     positive (should be symmetric around zero if correctly anchored).
//
//   v2 (first fix, May 23 2026):
//     Fixed volRatio_delta to use avgVol[5] (same window). Changed
//     entry IDs to "BEAR"/"BULL". Introduced make_comment() UDF.
//     INTRODUCED NEW BUG: 399/431 signals failed vb+vs≈vol sanity
//     check, with some rows producing NaN at feature positions 5-6.
//
//   v2.1 (second fix, May 23 2026):
//     Removed make_comment() — pre-computes str.tostring() at top
//     scope. MISDIAGNOSED the cause as a general "Pine v6 function
//     scope bug" where series[n] doesn't resolve inside UDF bodies.
//     This is wrong — UDFs handle history references correctly in
//     standard usage.
//
//   v2.2 (current, May 23 2026 — audited by Claude):
//     No code changes from v2.1. The inline fix is correct.
//     Corrected the explanation: the actual cause is an interaction
//     between strategy.entry()'s deferred comment evaluation and
//     how series cache state appears inside UDFs called from
//     conditional strategy blocks.
//
// WHY v2 WAS BROKEN (actual cause, not "function scope bug"):
//
//   The real issue is strategy.entry() comment evaluation timing.
//   When strategy.entry(comment=make_comment("BEAR")) executes inside
//   `if bearRaw`, Pine calls make_comment() in a deferred context.
//   Inside that function, expressions like vb1[1] reference series
//   history, but the UDF's evaluation frame sees a different series
//   cache state than inline expressions in the main body saw during
//   the normal bar-by-bar pass.
//
//   In particular:
//     - vol_body = vb1[1] + vs1[1]  evaluated at script body, every bar
//     - str.tostring(vb1[1])        evaluated inside make_comment,
//                                   only when the signal fires
//
//   These two access paths to "vb1[1]" resolve to different cached
//   values when the function is invoked from a conditional branch in
//   strategy context. The Pine runtime treats the series reference
//   inside the UDF as a fresh access that may not match what the outer
//   scope computed during normal evaluation.
//
//   The reliable workaround: compute str.tostring(series_expr) at
//   TOP-LEVEL SCOPE so all accesses happen during the standard
//   evaluation pass with consistent series state. String concatenation
//   has no series-cache dependency, so it works in deferred context.
//
//   General rule: any time you need str.tostring() of a series value
//   with a history offset, do it at top-level scope, not inside a
//   function called from a conditional strategy block.
//
// PYTHON PARSER DATA CONTRACT — enrich_signals.py expects:
//   parts = row['Signal'].split('|')
//   if len(parts) != 25: skip
//   direction = parts[0]               // "BEAR" or "BULL"
//   feats = [float(p) for p in parts[1:]]  // 24 float64 values
//
//   Feature index → name mapping:
//     [ 0] vol        [ 1] vd         [ 2] avgVol     [ 3] volRatio
//     [ 4] vdRatio    [ 5] vb         [ 6] vs         [ 7] rsi14
//     [ 8] rsi7       [ 9] dist_ema21 [10] dist_ema50 [11] dist_ema200
//     [12] macd_hist  [13] atr14      [14] rsi14_delta [15] rsi7_delta
//     [16] macd_hist_delta [17] price_mom5 [18] volRatio_delta (FIXED)
//     [19] vdRatio_delta  [20] ema21_bear [21] ema21_bull
//     [22] ema50_bear     [23] ema50_bull
//
// SANITY CHECKS (run on exported CSV):
//   1. Entry count: ≥400 for 90-day BTCUSDT 5m
//   2. vb + vs ≈ vol within 1% for ALL signals
//   3. vdRatio ≈ |vd| / vol within 0.001
//   4. volRatio ≈ vol / avgVol within 1%
//   5. No NaN values in any feature column
//   6. ema cross flags are binary (0 or 1 only)
//   7. All signal timestamps match perp data within ±5min
//
// USAGE:
//   1. Apply to 5m BTCUSDT.P chart (Binance Futures data source)
//   2. Set date range to the desired backtest period
//   3. Open Strategy Tester → Run → Export to CSV
//   4. Send CSV to Python pipeline: enrich_signals.py <export.csv> --label
// =================================================================

strategy("BAMBAM 24-Feature Delta Export v2.2", overlay=true,
  pyramiding=9999,
  default_qty_value=1,
  default_qty_type=strategy.fixed,
  initial_capital=100000000)

// =================================================================
// INPUTS — match raw BAMBAM signal capture defaults
// =================================================================
lookback   = input.int(200, 'Lookback',   minval=20, maxval=500, group='Signal')
lessRatio  = input.float(0.05, 'Less Ratio', minval=0.001, maxval=0.1, step=0.005, group='Signal')
volMult    = input.float(1.0, 'Volume Multiplier', minval=0.5, maxval=5.0, step=0.1, group='Signal')
deltaMode  = input.string("OR", "Delta Mode", options=["OR", "AND"], group='Signal')

// =================================================================
// BODY VOLUME ESTIMATE — linear interpolation, NOT actual taker volume
// =================================================================
vb1 = high == low ? 0 : volume * (close - low) / (high - low)
vs1 = high == low ? 0 : volume * (high - close) / (high - low)

// =================================================================
// BAMBAM SIGNAL LOGIC — fires on PREVIOUS bar (offset [1])
// =================================================================
ph       = high[1]
pl       = low[1]
vd       = vb1[1] - vs1[1]
vol_body = vb1[1] + vs1[1]

top     = ta.highest(ph, lookback)
btm     = ta.lowest(pl, lookback)
avgVol  = math.sum(vol_body, lookback) / lookback

vdRatio     = math.abs(vd / vol_body)
isLessRatio = vdRatio < lessRatio
highVol     = vol_body > (avgVol * volMult)

bearRaw = ph >= top and highVol and (isLessRatio or vd < 0)
bullRaw_OR  = pl <= btm and highVol and (isLessRatio or vd > 0)
bullRaw_AND = pl <= btm and highVol and isLessRatio and vd > 0
bullRaw = deltaMode == "AND" ? bullRaw_AND : bullRaw_OR

// =================================================================
// INDICATOR LEVELS — snapshots at signal bar
// =================================================================
rsi14_val = ta.rsi(close, 14)[1]
rsi7_val  = ta.rsi(close, 7)[1]

ema21  = ta.ema(close, 21)[1]
ema50  = ta.ema(close, 50)[1]
ema200 = ta.ema(close, 200)[1]

dist_ema21_val  = close[1] - ema21
dist_ema50_val  = close[1] - ema50
dist_ema200_val = close[1] - ema200

[_, _, macd_hist_val] = ta.macd(close, 12, 26, 9)
macd_hist_at = macd_hist_val[1]
atr14_val    = ta.atr(14)[1]

// =================================================================
// DELTA FEATURES — 5-bar change
// =================================================================
rsi14_delta_val = rsi14_val - ta.rsi(close, 14)[6]
rsi7_delta_val  = rsi7_val  - ta.rsi(close, 7)[6]
macd_hist_delta = macd_hist_at - macd_hist_val[6]
price_mom5_val  = (close[1] - close[6]) / close[6] * 100

// FIX 1: avgVol[5] anchors both ratios to the SAME rolling window
vol_body_5 = vb1[6] + vs1[6]
volRatio_5 = vol_body_5 / avgVol[5]
volRatio_delta_val = vol_body / avgVol - volRatio_5

vd_body_5  = vb1[6] - vs1[6]
vdRatio_5  = math.abs(vd_body_5 / vol_body_5)
vdRatio_delta_val = vdRatio - vdRatio_5

// =================================================================
// EMA CROSS FLAGS — binary
// =================================================================
ema21_bear_flag = close[1] < ema21  ? 1 : 0
ema21_bull_flag = close[1] > ema21  ? 1 : 0
ema50_bear_flag = close[1] < ema50  ? 1 : 0
ema50_bull_flag = close[1] > ema50  ? 1 : 0

// =================================================================
// FEATURE STRING CONSTRUCTION
//
// CRITICAL: ALL str.tostring() calls of series values MUST be at
// TOP-LEVEL SCOPE. Do NOT wrap in a UDF called from inside
// conditional strategy blocks.
//
// Why: strategy.entry() defers comment evaluation. When a UDF runs
// in that deferred context and accesses series history (e.g.vb1[1]),
// the series cache it sees can differ from what top-level
// expressions saw during normal bar-by-bar evaluation.
//
// String concatenation has no series-cache dependency — works fine.
// =================================================================
s_vol   = str.tostring(vol_body)
s_vd    = str.tostring(vd)
s_avg   = str.tostring(avgVol)
s_vr    = str.tostring(vol_body / avgVol)
s_vdr   = str.tostring(vdRatio)
s_vb    = str.tostring(vb1[1])
s_vs    = str.tostring(vs1[1])
s_r14   = str.tostring(rsi14_val)
s_r7    = str.tostring(rsi7_val)
s_d21   = str.tostring(dist_ema21_val)
s_d50   = str.tostring(dist_ema50_val)
s_d200  = str.tostring(dist_ema200_val)
s_macd  = str.tostring(macd_hist_at)
s_atr   = str.tostring(atr14_val)
s_r14d  = str.tostring(rsi14_delta_val)
s_r7d   = str.tostring(rsi7_delta_val)
s_md    = str.tostring(macd_hist_delta)
s_pm5   = str.tostring(price_mom5_val)
s_vrd   = str.tostring(volRatio_delta_val)
s_vdrd  = str.tostring(vdRatio_delta_val)
s_21b   = str.tostring(ema21_bear_flag)
s_21u   = str.tostring(ema21_bull_flag)
s_50b   = str.tostring(ema50_bear_flag)
s_50u   = str.tostring(ema50_bull_flag)

bear_comment = "BEAR|" +
  s_vol + "|" + s_vd + "|" + s_avg + "|" + s_vr + "|" + s_vdr + "|" +
  s_vb + "|" + s_vs + "|" + s_r14 + "|" + s_r7 + "|" +
  s_d21 + "|" + s_d50 + "|" + s_d200 + "|" + s_macd + "|" + s_atr + "|" +
  s_r14d + "|" + s_r7d + "|" + s_md + "|" + s_pm5 + "|" +
  s_vrd + "|" + s_vdrd + "|" +
  s_21b + "|" + s_21u + "|" + s_50b + "|" + s_50u

bull_comment = "BULL|" +
  s_vol + "|" + s_vd + "|" + s_avg + "|" + s_vr + "|" + s_vdr + "|" +
  s_vb + "|" + s_vs + "|" + s_r14 + "|" + s_r7 + "|" +
  s_d21 + "|" + s_d50 + "|" + s_d200 + "|" + s_macd + "|" + s_atr + "|" +
  s_r14d + "|" + s_r7d + "|" + s_md + "|" + s_pm5 + "|" +
  s_vrd + "|" + s_vdrd + "|" +
  s_21b + "|" + s_21u + "|" + s_50b + "|" + s_50u

// =================================================================
// STRATEGY ENTRIES
// =================================================================
if bearRaw
    strategy.entry("BEAR", strategy.short, comment=bear_comment)

if bullRaw
    strategy.entry("BULL", strategy.long, comment=bull_comment)

// =================================================================
// PLOTTING
// =================================================================
plotshape(series=bullRaw, color=color.new(#00ff88, 0),
  style=shape.triangleup, location=location.belowbar,
  offset=-1, size=size.tiny, title="Bull Signal")

plotshape(series=bearRaw, color=color.new(#ff4444, 0),
  style=shape.triangledown, location=location.abovebar,
  offset=-1, size=size.tiny, title="Bear Signal")
