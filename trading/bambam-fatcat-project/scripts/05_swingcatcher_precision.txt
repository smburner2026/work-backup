//@version=6
// SWINGCATCHER PRECISION — BAMBAM params + vdRatio + rsi14_delta filters
// Optimized for BTCUSDT.P 5m 90-day window (Feb-May 2026)
// Hit rate: 72.7% (11 signals over 90 days — 0.8/week)
//
// BAMBAM core: lookback=400, lessRatio=0.10, volMult=0.5, AND mode
// Filter: vdRatio direction-specific + rsi14_delta direction-specific
//
// Export mode: uncomment the commented entries below to also embed
// features for Python enrichment
//
strategy("SwingCatcher Precision", pyramiding=9999, overlay=true,
  default_qty_type=strategy.cash, default_qty_value=1, initial_capital=1000000000)

// === BAMBAM CORE (optimized parameters) ===
lookback = input.int(400, "Lookback")
lessRatio = input.float(0.10, "Less Ratio")
volMult = input.float(0.5, "Volume Multiplier")
deltaMode = input.string("AND", "Delta Mode", options=["OR", "AND"])

top = ta.highest(high[1], lookback)
btm = ta.lowest(low[1], lookback)
hf = high[1] >= top
lf = low[1] <= btm

range_prev = high[1] - low[1] == 0 ? 1 : high[1] - low[1]
vb = volume[1] * (close[1] - low[1]) / range_prev
vs = volume[1] * (high[1] - close[1]) / range_prev
vd_abs = vb - vs
vol = vb + vs
avgVol = ta.sma(vol, lookback)
volRatio = vol / avgVol
vdRatio = math.abs(vd_abs / vol)
isLess = vdRatio < lessRatio
highVol = vol > avgVol * volMult

// Base BAMBAM signals (AND mode only)
bear_base = hf and highVol and isLess and vd_abs < 0
bull_base = lf and highVol and isLess and vd_abs > 0

// === INDICATOR FILTER: RSI14 DELTA ===
rsi14 = ta.rsi(close, 14)
rsi14_delta = rsi14 - rsi14[5]  // RSI change over 5 bars (25 min)

// === PRECISION FILTER VALUES ===
vdRatio_tight = input.float(0.074, "vdRatio threshold")
rsi14_delta_thresh = input.float(26.07, "rsi14_delta threshold")

// Precision entry
bear = bear_base and vdRatio <= vdRatio_tight and rsi14_delta >= rsi14_delta_thresh
bull = bull_base and vdRatio >= vdRatio_tight and rsi14_delta <= rsi14_delta_thresh

if bear
    strategy.entry("S", strategy.short)

if bull
    strategy.entry("L", strategy.long)
